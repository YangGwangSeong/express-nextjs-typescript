## backend

```bash
$ npm init
$ npx tsc --init

$ npm install morgan nodemon express --save
$ npm install typescript ts-node
$ npm install @types/node @types/express @types/morgan --save-dev
$ npm install pg typeorm reflect-metadata --save

$ npx typeorm init

$ npm install bcryptjs class-validator class-transformer --save
$ npm install @types/bcryptjs --save-dev
$ npm install cors                                                              // CORS : Cross Origin Resource Share
$ npm i --save-dev @types/cors

$ npm install jsonwebtoken                                                      // jwt
$ npm i --save-dev @types/jsonwebtoken
$ npm install dotenv                                                            // 환경변수
$ npm install cookie                                                            // cookie
$ npm i --save-dev @types/cookie
```
